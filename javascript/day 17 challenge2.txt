<!DOCTYPE html>

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>IIFE Closure Example</title>
</head>

<body>

  <body style="background-color: rgb(24, 218, 72);"></body>
  <h1>A Closer Look at Function!</h1>

  <script>
    (function () {
      const header = document.querySelector('h1'); // Select once

      document.querySelector('body').addEventListener('click', function () {
        header.style.color = 'blue'; // Closure keeps reference to header
      });
    })();
  </script>
</body>
